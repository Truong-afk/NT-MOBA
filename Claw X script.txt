    TabWidth = 160,

-- Tạo button icon nhỏ khi minimize
local icon = Instance.new("ImageButton")
icon.Name = "RestoreIcon"
icon.Parent = game.Players.LocalPlayer.PlayerGui
icon.Size = UDim2.new(0, 50, 0, 50) -- Kích thước
icon.Position = UDim2.new(0.5, -25, 0.5, -25) -- Vị trí trên màn hình
icon.BackgroundTransparency = 1
icon.Image = "rbxassetid://122376714508463" -- Thay bằng ID asset của icon bạn muốn dùng
icon.Visible = false -- Mặc định ẩn, chỉ hiện khi giao diện bị minimize

-- Khi nhấn phím để minimize
local Minimized = false
local function ToggleWindow()
    Minimized = not Minimized
    Window:SetVisible(not Minimized) -- Ẩn hoặc hiện giao diện chính
    icon.Visible = Minimized -- Hiển thị hoặc ẩn icon
end
print("Icon đã được tạo và thêm vào PlayerGui")

-- Lắng nghe phím minimize
game:GetService("UserInputService").InputBegan:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.LeftControl then -- Phím minimize
        ToggleWindow()
    end
end)

-- Khi nhấn vào icon, giao diện khôi phục
icon.MouseButton1Click:Connect(function()
    ToggleWindow()
end)
