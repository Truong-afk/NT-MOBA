local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()

-- Tạo cửa sổ chính
local Window = Fluent:CreateWindow({
    Title = "NT MOBA",
    SubTitle = " | Tổng Hợp Script Ver1",
    TabWidth = 160,
    Size = UDim2.fromOffset(480, 360),
    Acrylic = false, -- Không dùng hiệu ứng mờ
    Theme = "Dark", -- Giao diện nền đen
    MinimizeKey = Enum.KeyCode.LeftControl -- Phím thu gọn
})

-- Thông báo khởi động
Fluent:Notify({
    Title = "NT MOBA",
    Content = "Chào mấy ní!",
    SubContent = "hihi",
    Duration = 5
})

-- Chức năng thu gọn toàn bộ menu
local isMinimized = false
local menuIcon = Instance.new("ImageButton") -- Biểu tượng menu nhỏ gọn
menuIcon.Image = "rbxassetid://1234567" -- Thay ID này bằng icon bạn muốn
menuIcon.Size = UDim2.new(0, 50, 0, 50)
menuIcon.Position = UDim2.new(0, 10, 0, 10) -- Vị trí trên màn hình
menuIcon.BackgroundTransparency = 1
menuIcon.Visible = false
menuIcon.Parent = game.CoreGui

Window:SetMinimizeCallback(function()
    isMinimized = not isMinimized
    if isMinimized then
        Window.Instance.Visible = false
        menuIcon.Visible = true
    else
        Window.Instance.Visible = true
        menuIcon.Visible = false
    end
end)

menuIcon.MouseButton1Click:Connect(function()
    isMinimized = false
    Window.Instance.Visible = true
    menuIcon.Visible = false
end)
